\begin{frame}{Transformer load flow calculation}
  \begin{columns}[c,onlytextwidth]
    \begin{column}{0.54\textwidth}
      \begin{itemize}
        \setlength{\itemsep}{0.5em}
        \item The power rating for network winding (NW) and converter
              winding (CW) are similar for VSC-HVDC transformers.
        \item Main inductance is very high and can be neglected for load
              flow calculations.
      \end{itemize}
    \end{column}

\begin{frame}{Transformer load flow calculation --- T-model}
  \begin{columns}[c,onlytextwidth]
    \begin{column}{0.54\textwidth}
      \begin{itemize}
        \setlength{\itemsep}{0.5em}
        \item For load flow calculations the transformers can be represented
              with a T-model.
        \item Calculation is based on the two-port representation of the
              transformer.
      \end{itemize}
    \end{column}
