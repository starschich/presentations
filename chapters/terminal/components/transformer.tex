\intermezzo{Converter transformer}{Connecting the ac system and the converter}

\begin{frame}{Converter transformer}
  \begin{columns}[c,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \setlength{\itemsep}{0.45em}
        \item Provides the connection between the ac system and the converter.
        \item Basic function as a normal ac power transformer: Transform the
              ac voltage between the ac grid level and the voltage required
              by the converter.
      \end{itemize}

\begin{frame}{On-Load Tap Changer (OLTC)}
  \begin{columns}[c,onlytextwidth]
    \begin{column}{0.56\textwidth}
      \begin{itemize}
        \setlength{\itemsep}{0.45em}
        \item Mechanical switching device that changes the turns-ratio of
              the transformer in voltage steps.
        \item A tap-changer step adds or subtracts 1--1.25\% to the
              converter transformer turns ratio.
        \item Usually up to $\sim$32 steps, giving a maximum voltage change
              of around 40\%.
        \item Mechanical device with limited operation speed --- cannot be
              used for dynamic control.
      \end{itemize}
      \vspace{0.3em}

    \end{column}
