\intermezzo{Pre-Insertion resistor (PIR)}{Energization current limitation}

\begin{frame}{Pre-Insertion resistor (PIR)}
  \begin{columns}[c,onlytextwidth]
    \begin{column}{0.56\textwidth}
      \begin{itemize}
        \setlength{\itemsep}{0.5em}
        \item Connecting capacitors to the AC grid would cause a high
              charging current into the converter, limited only by the
              transformer and converter impedance.
        \item The high current would cause high stress to the AC grid and
              the converter equipment.
        \item The PIR limits the current during the energization phase and
              is bypassed after the energization process has finished.
      \end{itemize}
    \end{column}
