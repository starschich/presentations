\begin{frame}{Measuring transducers}
  \begin{itemize}
    \setlength{\itemsep}{0.5em}
    \item Measure the analog voltage and current signals required for
          monitoring and control of the HVDC system.
    \item Located in the HV plant itself; usually convert the measured
          signal to a level and form suitable for subsequent processing
          ($<5$\,A, $<10$\,V).
    \item Any major scaling of the measured signal is implemented on the
          measurement device itself or on additional associated equipment.
    \item Provide galvanic (electrical) isolation between the raw HV signals
          and the clean environment of the control cubicles.
  \end{itemize}
\end{frame}

\begin{frame}{AC current measurement}
  \begin{columns}[c,onlytextwidth]
    \begin{column}{0.54\textwidth}
      \textbf{AC current transformer}
      \begin{itemize}
        \setlength{\itemsep}{0.4em}
        \item Measuring principle usable for AC current measurements only.
        \item Different insulators can be used: SF$_6$ or ``Clean Air''.
        \item Lacks the capability to measure currents at low frequencies
              ($<5$\,Hz).
      \end{itemize}
    \end{column}

\begin{frame}{DC current measurement}
  \begin{columns}[c,onlytextwidth]
    \begin{column}{0.54\textwidth}
      \textbf{DC Current Transducer (DCCT)} used for DC current measurement:
      \begin{itemize}
        \setlength{\itemsep}{0.35em}
        \item Zero Flux Current Transformers (ZFCT) or Optical Current Transformers (OCT)
        \item Measurement accuracy of 0.2\% or better
        \item Wide frequency range for very fast protection schemes
        \item ZFCT installation over a bushing or cable
      \end{itemize}
    \end{column}
