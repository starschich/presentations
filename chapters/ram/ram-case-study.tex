\intermezzo{Three-terminal hybrid HVDC interconnector for offshore wind}{Meisingset et al., CIGRE 2026, Paper B4 PS1 11874}


\begin{frame}{Study assumptions\\Scope \& method}
  \begin{block}{Method}
    Monte Carlo RAM analysis (1000 simulations) on Reliability Block Diagrams; energy availability (EA) based on \emph{available transmission capacity}.
  \end{block}
  \vspace{0.2em}
  \begin{itemize}
    \setlength{\itemsep}{0.35em}
    \item VSC base cases: SMP, RBP and BIP+DMR
    \item Half-bridge MMC with disconnectors only $\rightarrow$ No selective DC fault clearance.
    \item Model boundary:\\
    Offshore incl. 132\,kV GIS \& bays (excl. inter-array cables/turbines)\\
    Onshore incl. 420\,kV feeder (busbar excluded).
  \end{itemize}
\end{frame}

\begin{frame}{Study assumptions\\Reliability data \& repair}
  \begin{itemize}
    \setlength{\itemsep}{0.35em}
    \item \textbf{Converter:}
      \begin{itemize}
            \item Submodule spares 5-7
            \item Submodule failure rate (1\%/yr)
      \end{itemize}
      \item \textbf{Transformer}
      \begin{itemize}
            \item No spare parts for offshore transformer (delivery time 3 years)
            \item Offshore transformer MTBF 335\,y
      \end{itemize}
    \item \textbf{Scheduled maintenance (biennial):}
    \begin{itemize}
      \item SMP 6 days
      \item RBP/BIP 3.5 days per pole $+$ 8\,h neutral\\(Maintenance on the neutral area requires full shutdown).
    \end{itemize}
      
    \item \textbf{Outage on trip:} Onshore 2\,h
  \end{itemize}
\end{frame}

\begin{frame}{Study assumptions\\Reliability data \& repair}
  \begin{itemize}
    \setlength{\itemsep}{0.35em}
    \item \textbf{Cables MTTR:}
      \begin{itemize}
            \item Offshore cable: 75 d
            \item Onshore cable: 24 d
      \end{itemize}
      \item \textbf{Failure rates}
      \begin{itemize}
            \setlength{\itemsep}{1em}
            \item Bundled HVDC cable installation for SMP or RBIP:\qquad$0.01 \frac{\text{Failure}}{100km\cdot\text{Year}}$
            \item Bundled HVDC cable installation for BIP with DMR:\qquad$0.01 \frac{\text{Failure}}{100km\cdot\text{Year}}$
            \item Unbundled HVDC cable installation for BIP with DMR:\quad$0.005 \frac{\text{Failure}}{100km\cdot\text{Year}}$
      \end{itemize}
  \end{itemize}
\end{frame}
